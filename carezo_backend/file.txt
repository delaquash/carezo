# Start PostgreSQL and Redis using Docker
docker-compose up -d postgres redis

# Verify containers are running
docker ps

# You should see:
# driveease_postgres
# driveease_redis


# Install migrate tool
go install -tags 'postgres' github.com/golang-migrate/migrate/v4/cmd/migrate@latest

# Run migrations
migrate -path ./migrations \
  -database "postgresql://driveease_user:your_password_here@localhost:5432/driveease_db?sslmode=disable" \
  up

# Verify tables were created
docker exec -it driveease_postgres psql -U driveease_user -d driveease_db -c "\dt"

# You should see:
# users
# cars
# drivers
# bookings
# reviews
# sessions


# Run the application
go run cmd/api/main.go

# You should see:
# 🚀 Starting DriveEase in development mode
# ✅ Connected to PostgreSQL successfully
# ✅ Connected to Redis successfully
# 🌐 Server listening on port 8080
